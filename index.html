  <script>

    const stage = document.getElementById("stage");

    const noBtn  = document.getElementById("no");

    const yesBtn = document.getElementById("yes");

    const mainScreen = document.getElementById("mainScreen");

    const finalScreen = document.getElementById("finalScreen");

    // Form bits

    const datePick = document.getElementById("datePick");

    const timePick = document.getElementById("timePick");

    const note = document.getElementById("note");

    const makeMsg = document.getElementById("makeMsg");

    const copyMsg = document.getElementById("copyMsg");

    const output = document.getElementById("output");

    const reset = document.getElementById("reset");

    const toast = document.getElementById("toast");

    // Difficulty tuning

    let dodges = 0;

    let noScale = 1;

    let dodgeCooldown = 260;

    function rand(min, max){ return Math.floor(Math.random()*(max-min+1))+min; }

    function placeNoRandom(){

      const s = stage.getBoundingClientRect();

      const b = noBtn.getBoundingClientRect();

      const pad = 12;

      const maxX = s.width - b.width - pad;

      const maxY = s.height - b.height - pad;

      const x = rand(pad, Math.max(pad, maxX));

      const y = rand(pad, Math.max(pad, maxY));

      noBtn.style.left = `${x}px`;

      noBtn.style.top = `${y}px`;

      noBtn.style.bottom = "auto";

    }

    let lastMove = 0;

    function dodge(){

      const now = Date.now();

      if (now - lastMove < dodgeCooldown) return;

      lastMove = now;

      dodges++;

      noScale = Math.max(0.45, 1 - dodges * 0.06);

      noBtn.style.transform = `scale(${noScale})`;

      dodgeCooldown = Math.max(80, 260 - dodges * 18);

      if (dodges === 3) noBtn.textContent = "No?? 😭";

      if (dodges === 6) noBtn.textContent = "Stoppp 😂";

      if (dodges === 9) noBtn.textContent = "Okay fine 😵";

      placeNoRandom();

    }

    // Desktop

    noBtn.addEventListener("mouseenter", dodge);

    // Mobile

    noBtn.addEventListener("touchstart", (e) => {

      e.preventDefault();

      dodge();

    }, { passive:false });

    noBtn.addEventListener("click", (e) => {

      e.preventDefault();

      dodge();

    });

    stage.addEventListener("touchmove", () => dodge(), { passive:true });

    // YES handler -> show final screen + confetti

    yesBtn.addEventListener("click", () => {

      mainScreen.style.display = "none";

      finalScreen.classList.add("show");

      startConfetti(2200);

      // small UX: auto-focus date

      setTimeout(() => datePick.focus(), 300);

    });

    setTimeout(placeNoRandom, 350);

    /* --------- Date message generator ---------- */

    function formatDate(yyyy_mm_dd){

      if(!yyyy_mm_dd) return "";

      const [y,m,d] = yyyy_mm_dd.split("-").map(Number);

      const dt = new Date(y, m-1, d);

      // Simple readable format

      return dt.toLocaleDateString(undefined, { weekday:"short", day:"2-digit", month:"short" });

    }

    function formatTime(hh_mm){

      if(!hh_mm) return "";

      let [h, m] = hh_mm.split(":").map(Number);

      const ampm = h >= 12 ? "PM" : "AM";

      h = h % 12; if (h === 0) h = 12;

      return `${h}:${String(m).padStart(2,"0")} ${ampm}`;

    }

    function buildMessage(){

      const d = formatDate(datePick.value);

      const t = formatTime(timePick.value);

      const when =

        (d && t) ? `${d} at ${t}` :

        (d && !t) ? `${d}` :

        (!d && t) ? `${t}` :

        "";

      const extra = note.value.trim();

      const base =

`She said YES 💘

Valentine date: ${when || "TBD 😌"}

${extra ? "Note: " + extra : ""}

Now confirm. No backing out 😂`;

      return base.trim();

    }

    makeMsg.addEventListener("click", () => {

      const msg = buildMessage();

      output.textContent = msg;

      output.classList.add("show");

    });

    function showToast(text){

      toast.textContent = text;

      toast.classList.add("show");

      setTimeout(() => toast.classList.remove("show"), 900);

    }

    copyMsg.addEventListener("click", async () => {

      const msg = buildMessage();

      output.textContent = msg;

      output.classList.add("show");

      try{

        await navigator.clipboard.writeText(msg);

        showToast("Copied ✨");

      } catch(e){

        // fallback for older browsers

        const ta = document.createElement("textarea");

        ta.value = msg;

        document.body.appendChild(ta);

        ta.select();

        document.execCommand("copy");

        ta.remove();

        showToast("Copied ✨");

      }

    });

    reset.addEventListener("click", () => {

      // reset to first screen

      finalScreen.classList.remove("show");

      mainScreen.style.display = "block";

      output.classList.remove("show");

      output.textContent = "";

      datePick.value = "";

      timePick.value = "";

      note.value = "";

      // reset No difficulty

      dodges = 0;

      noScale = 1;

      dodgeCooldown = 260;

      noBtn.textContent = "No 🙃";

      noBtn.style.transform = "scale(1)";

      noBtn.style.bottom = "18px";

      noBtn.style.top = "auto";

      noBtn.style.left = "150px";

    });

    /* ---------------- Confetti (no libraries) ---------------- */

    const canvas = document.getElementById("confetti");

    const ctx = canvas.getContext("2d");

    let W, H;

    function resize(){

      W = canvas.width = window.innerWidth;

      H = canvas.height = window.innerHeight;

    }

    window.addEventListener("resize", resize);

    resize();

    function startConfetti(durationMs){

      canvas.classList.add("show");

      const pieces = [];

      const count = 190;

      for (let i=0;i<count;i++){

        pieces.push({

          x: Math.random()*W,

          y: -20 - Math.random()*H*0.3,

          w: 6 + Math.random()*8,

          h: 8 + Math.random()*14,

          vx: -2 + Math.random()*4,

          vy: 3 + Math.random()*6,

          rot: Math.random()*Math.PI,

          vr: -0.2 + Math.random()*0.4,

          hue: Math.floor(Math.random()*360),

          alpha: 0.9

        });

      }

      const start = performance.now();

      function frame(t){

        const elapsed = t - start;

        ctx.clearRect(0,0,W,H);

        for (const p of pieces){

          p.x += p.vx;

          p.y += p.vy;

          p.rot += p.vr;

          p.vy += 0.02;

          if (p.x < -50) p.x = W+50;

          if (p.x > W+50) p.x = -50;

          ctx.save();

          ctx.translate(p.x, p.y);

          ctx.rotate(p.rot);

          ctx.fillStyle = `hsla(${p.hue}, 90%, 60%, ${p.alpha})`;

          ctx.fillRect(-p.w/2, -p.h/2, p.w, p.h);

          ctx.restore();

        }

        if (elapsed < durationMs){

          requestAnimationFrame(frame);

        } else {

          ctx.clearRect(0,0,W,H);

          canvas.classList.remove("show");

        }

      }

      requestAnimationFrame(frame);

    }
